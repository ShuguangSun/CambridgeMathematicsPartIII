  g(X_{i}, \theta_{0}))^{2})^{-\frac{1}{2}} - 2 (n^{-1} \sum \mu_{i}^{2})^{-\frac{1}{2}}]
\end{align}

On $\Theta^{\star} = [\theta_{0} - M, \theta_{0} + M]$, consider $c(x)

\end{align} and
\begin{align}
